def folding(array,timeperiod):
    """Returns the folded time array"""
    
    tarray = []
    
    for x in array:
        if x < timeperiod:
            tarray.append(x)
        else:
            while x > timeperiod:
                x -= timeperiod
            tarray.append(x)
            
    return tarray
